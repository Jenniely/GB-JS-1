1. Дан код. Почему код даёт именно такие результаты?

var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2
d = b++; alert(d); // 1
c = (2+ ++a); alert(c); // 5
d = (2+ b++); alert(d); // 4
alert(a); // 3
alert(b); // 3

a увеличивается на 1, возвращается значение.

При использовании постфиксного инкремента сначала переменная используется в вызове, а потом увеличивается.

a = 2; при вычислении выражения в скобках увеличивается до 3 и складывается, возвращается результат всего выражения.

b = 2; при вычислении выражения в скобках складывается, возвращается результат всего выражения, потом b увеличивается на 1.

Итоговые значения a и b - 3.

  • если a и b положительные, вывести их разность;
  • если а и b отрицательные, вывести их произведение;
  • если а и b разных знаков, вывести их сумму; ноль можно считать положительным числом.
function numbersPlay() {
let result = document.getElementById('result_2-1');
let a = ((Math.random() < 0.5) ? -1 : 1)*Math.floor(Math.random() * (1000 - 1)) + 1;
let b = ((Math.random() < 0.5) ? -1 : 1)*Math.floor(Math.random() * (1000 - 1)) + 1;
let res = 0;
let operation = '';
if (a >= 0 && b >= 0) {
res = a - b;
operation = 'разность';
} else if (a < 0 && b < 0) {
res = a * b;
operation = 'произведение';
} else {
res = a + b;
operation = 'сумма';
}
}

function mathOperation(operation, a, b) {

case 'sum':

case 'substract':
return substractTwo(a,b);
case 'multiply':
return multiplyTwo(a,b);
